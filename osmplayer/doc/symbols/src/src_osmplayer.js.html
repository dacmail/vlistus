 98 /**
 99  * @see minplayer.plugin.construct
100  */
101 osmplayer.prototype.construct = function() {
102 
103   // Call the minplayer display constructor.
104   minplayer.prototype.construct.call(this);
105 
106   // We need to cleanup the player when it has been destroyed.
107   jQuery(this.display).bind('playerdestroyed', (function(player) {
108     return function(element) {
109       if (element === player.display.eq(0)[0]) {
110         for (var plugin in minplayer.plugins[player.options.id]) {
111           for (var index in minplayer.plugins[player.options.id][plugin]) {
112             minplayer.plugins[player.options.id][plugin][index].destroy();
113             delete minplayer.plugins[player.options.id][plugin][index];
114           }
115           minplayer.plugins[player.options.id][plugin].length = 0;
116         }
117         delete minplayer.plugins[player.options.id];
118         minplayer.plugins[player.options.id] = null;
119       }
120     };
121   })(this));
122 
123   /** The play queue and index. */
124   this.playQueue = [];
125   this.playIndex = 0;
126   this.hasPlaylist = false;
127 
128   /** The playlist for this media player. */
129   this.create('playlist', 'osmplayer');
130 
131   /** Get the playlist or any other playlist that connects. */
132   this.get('playlist', function(playlist) {
133     this.hasPlaylist = true;
134     playlist.ubind(this.uuid + ':nodeLoad', (function(player) {
135       return function(event, data) {
136         player.loadNode(data);
137       };
138     })(this));
139   });
140 
141   // Play each media sequentially...
142   this.get('media', function(media) {
143     media.ubind(this.uuid + ':ended', (function(player) {
144       return function() {
145         player.options.autoplay = true;
146         player.playNext();
147       };
148     })(this));
149   });
150 
151   // Load the node if one is provided.
152   if (this.options.node) {
153     this.loadNode(this.options.node);
154   }
155 };

166 /**
167  * Reset the osmplayer.
168  *
169  * @param {function} callback Called when it is done resetting.
170  */
171 osmplayer.prototype.reset = function(callback) {
172 
173   // Empty the playqueue.
174   this.playQueue.length = 0;
175   this.playQueue = [];
176   this.playIndex = 0;
177 
178   // Clear the playloader.
179   if (this.playLoader && this.options.preview) {
180     this.options.preview = '';
181     this.playLoader.clear((function(player) {
182       return function() {
183         callback.call(player);
184       };
185     })(this));
186   }
187   else if (callback) {
188     callback.call(this);
189   }
190 };

192 /**
193  * The load node function.
194  *
195  * @param {object} node A media node object.
196  * @return {boolean} If the node was loaded.
197  */
198 osmplayer.prototype.loadNode = function(node) {
199 
200   // Make sure this is a valid node.
201   if (!node || (node.hasOwnProperty('length') && (node.length === 0))) {
202     return false;
203   }
204 
205   // Reset the player.
206   this.reset(function() {
207 
208     // Set the hasMedia flag.
209     this.hasMedia = node && node.mediafiles && node.mediafiles.media;
210 
211     // If this node is set and has files.
212     if (node && node.mediafiles) {
213 
214       // Load the media files.
215       var media = node.mediafiles.media;
216       if (media) {
217         var file = null;
218         var types = [];
219 
220         // For mobile devices, we should only show the main media.
221         if (minplayer.isAndroid || minplayer.isIDevice) {
222           types = ['media'];
223         }
224         else {
225           types = ['intro', 'commercial', 'prereel', 'media', 'postreel'];
226         }
227 
228         // Iterate through the types.
229         jQuery.each(types, (function(player) {
230           return function(key, type) {
231             file = player.addToQueue(media[type]);
232             if (file) {
233               file.queueType = type;
234             }
235           };
236         })(this));
237       }
238       else {
239 
240         // Add a class to the display to let themes handle this.
241         this.display.addClass('nomedia');
242       }
243 
244       // Play the next media
245       this.playNext();
246 
247       // Load the preview image.
248       osmplayer.getImage(node.mediafiles, 'preview', (function(player) {
249         return function(image) {
250           if (player.playLoader && (player.playLoader.display.length > 0)) {
251             player.playLoader.enabled = true;
252             player.playLoader.loadPreview(image.path);
253             player.playLoader.previewFlag.setFlag('media', true);
254             if (!player.hasMedia) {
255               player.playLoader.busy.setFlag('media', false);
256               player.playLoader.bigPlay.setFlag('media', false);
257             }
258             player.playLoader.checkVisibility();
259           }
260         };
261       })(this));
262     }
263   });
264 };

266 /**
267  * Adds a file to the play queue.
268  *
269  * @param {object} file The file to add to the queue.
270  * @return {object} The file that was added to the queue.
271  */
272 osmplayer.prototype.addToQueue = function(file) {
273   file = minplayer.getMediaFile(file);
274   if (file) {
275     this.playQueue.push(file);
276   }
277   return file;
278 };

280 /**
281  * Plays the next media file in the queue.
282  */
283 osmplayer.prototype.playNext = function() {
284   if (this.playQueue.length > this.playIndex) {
285     this.load(this.playQueue[this.playIndex]);
286     this.playIndex++;
287   }
288   else if (this.options.repeat) {
289     this.playIndex = 0;
290     this.playNext();
291   }
292   else if (this.playQueue.length > 0) {
293 
294     // If we have a playlist, let them handle what to do next.
295     if (this.hasPlaylist && this.options.autoNext) {
296       this.trigger('player_ended');
297     }
298     else {
299       // If there is no playlist, and no repeat, we will
300       // just seek to the beginning and pause.
301       this.options.autoplay = false;
302       this.playIndex = 0;
303       this.playNext();
304     }
305   }
306   else if (this.media) {
307     this.media.stop();
308 
309     // If there is no media found, then clear the player.
310     if (!this.hasMedia) {
311       this.media.clear();
312     }
313   }
314 };

316 /**
317  * Returns a node.
318  *
319  * @param {object} node The node to get.
320  * @param {function} callback Called when the node is retrieved.
321  */
322 osmplayer.getNode = function(node, callback) {
323   if (node && node.mediafiles && node.mediafiles.media) {
324     var mediaFile = minplayer.getMediaFile(node.mediafiles.media.media);
325     if (mediaFile) {
326       var player = minplayer.players[mediaFile.player];
327       if (player && (typeof player.getNode === 'function')) {
328         player.getNode(mediaFile, function(node) {
329           callback(node);
330         });
331       }
332     }
333   }
334 };

336 /**
337  * Returns an image provided image array.
338  *
339  * @param {object} mediafiles The mediafiles to search within.
340  * @param {string} type The type of image to look for.
341  * @param {function} callback Called when the image is retrieved.
342  */
343 osmplayer.getImage = function(mediafiles, type, callback) {
344 
345   var image = '';
346   var images = mediafiles.image;
347   if (images) {
348 
349     // If the image type exists, then just use that one...
350     if (images[type]) {
351       image = images[type];
352     }
353     // Or try the original image...
354     else if (images.image) {
355       image = images.image;
356     }
357     // Otherwise, just try ANY image...
358     else {
359 
360       // Or, just pick the first one available.
361       for (type in images) {
362         if (images.hasOwnProperty(type)) {
363           image = images[type];
364           break;
365         }
366       }
367     }
368   }
369 
370   // If the image exists, then callback with that image.
371   if (image) {
372     callback(new minplayer.file(image));
373   }
374   else {
375     // Get the image from the media player...
376     var mediaFile = minplayer.getMediaFile(mediafiles.media.media);
377     if (mediaFile) {
378       var player = minplayer.players[mediaFile.player];
379       if (player && (typeof player.getImage === 'function')) {
380         player.getImage(mediaFile, type, function(src) {
381           callback(new minplayer.file(src));
382         });
383       }
384     }
385   }
386 };
