 35 /**
 36  * @see minplayer.players.base#canPlay
 37  *
 38  * @param {object} file A {@link minplayer.file} object.
 39  * @return {boolean} If this player can play this media type.
 40  */
 41 minplayer.players.limelight.canPlay = function(file) {
 42 
 43   // Check for the mimetype for limelight.
 44   if (file.mimetype === 'video/limelight') {
 45     return true;
 46   }
 47 
 48   // If the path is a limelight path, then return true.
 49   var regex = /.*limelight\.com.*/i;
 50   return (file.path.search(regex) === 0);
 51 };
 52 
 53 /**
 54  * Return the ID for a provided media file.
 55  *
 56  * @param {object} file A {@link minplayer.file} object.
 57  * @return {string} The ID for the provided media.
 58  */
 59 minplayer.players.limelight.getMediaId = function(file) {
 60   var regex = /.*limelight\.com.*mediaId=([a-zA-Z0-9]+)/i;
 61   if (file.path.search(regex) === 0) {
 62     return file.path.match(regex)[1];
 63   }
 64   else {
 65     return file.path;
 66   }
 67 };

 69 /**
 70  * Register this limelight player so that multiple players can be present
 71  * on the same page without event collision.
 72  */
 73 minplayer.players.limelight.prototype.register = function() {
 74 
 75   // Register the limelight player.
 76   window.delvePlayerCallback = function(playerId, event, data) {
 77 
 78     // Get the main player ID.
 79     var id = playerId.replace('-player', '');
 80 
 81     // Dispatch this event to the correct player.
 82     jQuery.each(minplayer.get(id, 'media'), function(key, media) {
 83       media.onMediaUpdate(event, data);
 84     });
 85   };
 86 };

 88 /**
 89  * The media update method.
 90  *
 91  * @param {string} event The event that was triggered.
 92  * @param {object} data The event object.
 93  */
 94 minplayer.players.limelight.prototype.onMediaUpdate = function(event, data) {
 95 
 96   // Switch on the event name.
 97   switch (event) {
 98     case 'onPlayerLoad':
 99       this.onReady();
100       break;
101 
102     case 'onMediaLoad':
103       // If this media has already completed, then pause it and return...
104       if (this.complete) {
105         this.pause();
106         this.onPaused();
107         return;
108       }
109 
110       this.shouldSeek = (this.getSeek() > 0);
111       this.onLoaded();
112       break;
113 
114     case 'onMediaComplete':
115       this.complete = true;
116       this.onComplete();
117       break;
118 
119     case 'onPlayheadUpdate':
120 
121       // Make sure we say this is playing.
122       if (data.positionInMilliseconds && !this.playing && !this.complete) {
123         this.onPlaying();
124       }
125 
126       // Set the complete flag to false.
127       this.complete = false;
128 
129       // Set the duration and current time.
130       if (this.shouldSeek && this.seekValue) {
131         this.shouldSeek = false;
132         this.seek(this.seekValue);
133       }
134       else {
135         this.duration.set(data.durationInMilliseconds / 1000);
136         this.currentTime.set(data.positionInMilliseconds / 1000);
137       }
138       break;
139 
140     case 'onError':
141       this.onError();
142       break;
143 
144     case 'onPlayStateChanged':
145       if (data.isPlaying) {
146         this.onPlaying();
147       }
148       else if (data.isBusy) {
149         this.onWaiting();
150       }
151       else {
152         this.onPaused();
153       }
154       break;
155   }
156 };

158 /**
159  * @see minplayer.players.base#create
160  * @return {object} The media player entity.
161  */
162 minplayer.players.limelight.prototype.createPlayer = function() {
163   minplayer.players.flash.prototype.createPlayer.call(this);
164 
165   // Insert the embed.js.
166   var tag = document.createElement('script');
167   tag.src = 'https://assets.delvenetworks.com/player/embed.js';
168   var firstScriptTag = document.getElementsByTagName('script')[0];
169   firstScriptTag.parentNode.insertBefore(tag, firstScriptTag);
170 
171   // Now register this player.
172   this.register();
173 
174   // Get the FlashVars.
175   var flashVars = {
176     'deepLink': 'true',
177     'autoplay': this.options.autoplay ? 'true' : 'false',
178     'startQuality': 'HD'
179   }, regex = null;
180 
181   // Get the channel for this player.
182   var channel = this.options.channel;
183   if (!channel) {
184     regex = /.*limelight\.com.*channelId=([a-zA-Z0-9]+)/i;
185     if (this.mediaFile.path.search(regex) === 0) {
186       channel = this.mediaFile.path.match(regex)[1];
187     }
188   }
189 
190   // Set the channel.
191   if (channel && this.mediaFile.queueType === 'media') {
192     flashVars.adConfigurationChannelId = channel;
193   }
194 
195   // Get the playerForm for this player.
196   var playerForm = this.options.playerForm;
197   if (!playerForm) {
198     regex = /.*limelight\.com.*playerForm=([a-zA-Z0-9]+)/i;
199     if (this.mediaFile.path.search(regex) === 0) {
200       playerForm = this.mediaFile.path.match(regex)[1];
201     }
202   }
203 
204   // Set the player form.
205   if (playerForm) {
206     flashVars.playerForm = playerForm;
207   }
208 
209   // Add the media Id to the flashvars.
210   flashVars.mediaId = this.mediaFile.id;
211 
212   // Set the player ID.
213   var playerId = this.options.id + '-player';
214 
215   // Check the embed code every second.
216   setTimeout(function checkLimelight() {
217     if (window.hasOwnProperty('LimelightPlayerUtil')) {
218       window.LimelightPlayerUtil.initEmbed(playerId);
219     }
220     else {
221       setTimeout(checkLimelight, 1000);
222     }
223   }, 1000);
224 
225   // Return a flash media player object.
226   return this.getFlash({
227     swf: 'https://assets.delvenetworks.com/player/loader.swf',
228     id: playerId,
229     width: this.options.width,
230     height: '100%',
231     flashvars: flashVars,
232     wmode: this.options.wmode
233   });
234 };
